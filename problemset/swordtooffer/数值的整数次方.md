## 题目描述
给定一个double类型的浮点数base和int类型的整数exponent。求base的exponent次方。

```c++
class Solution {
public:
    double Power(double base, int exponent) {
        double res = 0;
        int exp = 1,flag = 1;
        if (exponent < 0){
            exponent = -exponent;
            flag = 0;
        }
        if (exponent == 0){
            res = 1;
        }else if (exponent == 1){
            res = base;
        }else{ 
            // 快速幂
            int nbase = base;
            while(exponent > exp*2){
                res = nbase*nbase;
                nbase = res;
                exponent -= exp*2;
                exp++;
            }
            for (int i = 0;i < exponent;++i){
                res *= base;
            }
        }
        return flag == 1?res:1/res;
    }
};
```
