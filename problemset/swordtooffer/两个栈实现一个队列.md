## 题目描述
用两个栈来实现一个队列，完成队列的Push和Pop操作。 队列中的元素为int类型。

```c++
class Solution
{
public:
    void push(int node) {
        stack1.push(node);
    }

    int pop() {
        if (stack2.size() == 0){
            while(stack1.size() > 0){
                int item = stack1.top();
                stack1.pop();// pop没有返回值
                stack2.push(item);
            }
        }
        int res = stack2.top();
        stack2.pop();// 先top()获得栈顶元素，再调用pop()弹出栈顶元素
        return res;
    }

private:
    stack<int> stack1;
    stack<int> stack2;
};
```
