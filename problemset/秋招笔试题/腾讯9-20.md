### 1 电话号码
```C++
#include <iostream>
using namespace std;
int main() {
    int t,n;
    string s;
    cin>>t;
    while (t--) {
        cin>>n>>s;
        bool yes = false;
        if (n >= 11){
            for (int i = 0;i <= n - 11;i++){
                if (s[i] == '8') {
                    yes = true;
                    break;
                }
            }
        }
        cout<<(yes?"YES":"NO")<<endl;
    }
    return 0;
}
```

### 3 圆圈
```c++
#include <iostream>
#include <algorithm>
using namespace std;
int a[101];
int res[101];
int main() {
    int n;
    cin>>n;
    for (int i = 0;i < n;i++) {
        cin>>a[i];
    }
    sort(a,a+n);
    int k = 0;
    for (int i = 0;i < n/2;i++){
        res[i] = a[k];
        res[n-i-1] = a[k+1];
        k+=2;
    }
    if (n % 2 == 1) {
        res[n/2] = a[n/2];
    }
    for (int i = 0;i < n;i++){
        cout<<res[i];
        if (i != n - 1) cout<<" "<<endl;
    }
    
    cout<<endl;
    return 0;
}
```

### 4刷漆

```C++
#include <iostream>
using namespace std;
int a[5001];
int main() {
    int n;
    cin>>n;
    long long max = 0;
    for (int i = 0;i < n;i++) {
        cin>>a[i];
        if (a[i] > max) {
            max = a[i];
        }
    }
    int count = 1;
    if (max*n/2 < n){
        for (int i = 1;i < max;i++) {
            for (int j = 0;j < n;j++){
                if (a[j] < i+1){
                    count++;
                }
            }
        }
    }else {
        count = n;
    }
    
    cout<<(count>n?count:n)<<endl;
    return 0;
 }
```
